package maze.logic;

/**
 *
 * @author Tiago
 */
public class GameStatus {
    
    private Board board;
    private boolean gameOver;
    
    public GameStatus() {
        gameOver = false;
    }
    
    public void initializeBoard() {
        board = new Board();
    }
    
    public void initializeBoard(char[][] maze) {
        board = new Board(maze);
    }
    
    public boolean gameOver() {
        return gameOver;
    }
    
    public Board getBoard() {
        return board;
    }
    
    public String toString(Object object) {
        return board.toString();
    }
    
    public void updateGameStatus(GameMode mode, String direction) {
        boolean heroOnExit = false;
        
        board.updateBoard(mode, direction);
        
        if(board.getHero().getPoint().getX() == board.getExit()[0] && board.getHero().getPoint().getY() == board.getExit()[1])
            heroOnExit = true;
        
        if(board.getHero().isDead() || 
                (board.getDragon().isDead() && heroOnExit)) gameOver = true;
    }
}
